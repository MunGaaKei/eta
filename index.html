ETA

介绍 基本 eta.use 数据控制

介绍

基于 Web Components 标准，把原生HTML封装为灵活且可复用的组件。

#基本

通过 <template> 编写

页面加载完后会自动根据 template 标签的 name 属性值注册组件。



/* use component */
<i-input btn="提交"></i-input>

/* component template */
<template name="i-input">
<input type="text" class="i-input">
<button class="i-input-btn">#btn#</button>
<style>
    i-input {
        display: flex;
    }
    .i-input {
        padding: 4px 8px;
        background: #eee;
        outline: none;
    }
    .i-input-btn {
        padding: 4px 8px;
        background: #bfceff;
    }
</style>
<script>
eta.set('i-input', {
    handleSubmit (e) {
        console.log(this, e);
    }
})
</script>
</template>

#eta.use( component, tagName )

组件引入



<i-banner title="BANNER"></i-banner>
            

import banner from 'banner.js';

eta.use(banner, 'i-banner');
            

/* banner.js */
export default {
    template: `<div class="i-banner">
        <b>#title#</b>
    </div>
    
    <style>
    .i-banner {
        text-align: center;
        line-height: 100px;
        font-size: 32px;
        background: black;
        color: white;
    }
    </style>`,
}
